מערכת דיוור וקמפיינים
הנהלה

זוהי מערכת הדיוור השיווקי של WHITEBOARD עצמה (ברמת ההנהלה) — לא מודול למרצים. מטרתה לטפח (nurture) ולהמיר את הלידים של הפלטפורמה: לסגמנט אותם לפי קבוצות ותגיות, לשלוח קמפיינים בודדים, ולבנות סדרות אוטומציה מבוססות טריגר. המערכת עושה שימוש חוזר מלא בעורך המייל הויזואלי הקיים (אותו עורך בלוקים של מודול הדיוור למרצים) — לא נבנה עורך שני.

כללי יסוד למימוש

  1. שימוש חוזר מלא בעורך הבלוקים הקיים (עורך מייל ויזואלי) — אותו מודל בלוקים, אותו registry. אסור לשכפל עורך.
  2. Snapshot של משתנים בשליחה — כל ערכי המשתנים ({{first_name}} וכו') נדגמים ברגע השליחה, לא בזמן העריכה.
  3. בלוק הסרה מרשימת התפוצה חובה — כל מייל חייב לכלול בלוק "הסרה מרשימת התפוצה" (חובה חוקית), מוזרק כברירת מחדל.
  4. תיוג אוטומטי event-driven — רץ על אירועים, לא polling.
  5. סגמנט = שאילתה דינמית (לא רשימה קפואה); קהל שמור יכול להיות דינמי או snapshot.

01סקירה ומבנה

המודול מאפשר להנהלה לסגמנט את הלידים לפי קבוצות חופשיות (free-form) ותגיות, לשלוח קמפיינים בודדים, ולבנות סדרות מבוססות טריגר. כל המסכים יושבים תחת מרכז בקרה הנהלה → שיווק ומכירות.

דשבורד דיוור

שליחות, פתיחות, הקלקות, ביצועי קמפיינים וסדרות.

רשימות דיוור

רשימות ידניות (snapshot) בשמות חופשיים + יוצר רשימה לפי תויות. סגמנטים דינמיים — שלב 2.

קמפיין בודד

בחירת קהל → עריכת מייל → תזמון ושליחה.

בנאי סדרות

טריגר + צעדים (מייל / המתנה / תנאי) באוטומציה.

שיווק ומשפך קיים

מסך קיים שמוביל לפעולת "צור קמפיין" על סגמנט לידים.

אינטגרציה: מסך "שיווק ומשפך" הקיים אינו מוחלף — הוא מקבל פריט ניווט "דיוור וקמפיינים" וכפתור "קמפיין חדש" שפותח את זרימת הקמפיין הבודד.

מה מאופיין בפועל (שלב 1) מול עתידי: ה‑UID שנבנה כולל רשימות דיוור ידניות (snapshot), קמפיין בודד (אשף 4 שלבים עם בחירת בלוקים/HTML), סדרות עם פיצול‑תנאי לשני מסלולים ואנליטיקת שלב. סעיפי סגמנטים דינמיים (סעיף 04) ומנוע התיוג האוטומטי (סעיף 03) מתוארים כאן כ‑שלב 2 — הם עדיין לא בממשק.

02מודל הנתונים (DB Schema)

הסכמה מתבססת על contacts ו‑tags שכבר משותפים עם ה‑CRM ("כל אנשי הקשר") — מקור אמת אחד. מעליהם נבנים תיוג, סגמנטים, קהלים, קמפיינים, סדרות וטבלת השליחות.

-- ── אנשי קשר ותגיות (משותף עם CRM — מקור אמת יחיד) ──
CREATE TABLE contacts (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  email       text NOT NULL UNIQUE,
  first_name  text,
  last_name   text,
  phone       text,
  source      text,                  -- מאיפה הגיע הליד
  status      text NOT NULL DEFAULT 'lead',  -- lead | trial | customer | churned
  created_at  timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE tags (
  id    uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name  text NOT NULL,
  kind  text NOT NULL DEFAULT 'manual',  -- 'manual' | 'auto'
  color text
);

CREATE TABLE contact_tags (
  contact_id uuid REFERENCES contacts(id) ON DELETE CASCADE,
  tag_id     uuid REFERENCES tags(id) ON DELETE CASCADE,
  added_by   text NOT NULL DEFAULT 'manual',  -- 'manual' | 'rule'
  rule_id    uuid,                          -- מי מה-tag_rules תייג
  created_at timestamptz NOT NULL DEFAULT now(),
  PRIMARY KEY (contact_id, tag_id)
);

-- ── מנוע התיוג האוטומטי ──
CREATE TABLE tag_rules (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name            text NOT NULL,
  trigger_event   text NOT NULL,    -- 'webinar_registered' → תייג 'וובינר'
  conditions_json jsonb,            -- תנאים נוספים על האירוע/הקונטקט
  tag_id          uuid REFERENCES tags(id),
  active          boolean NOT NULL DEFAULT true
);

-- ── סגמנטים (הגדרת פילטר דינמי) וקהלים שמורים ──
CREATE TABLE segments (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name            text NOT NULL,
  logic           text NOT NULL DEFAULT 'and',  -- 'and' | 'or'
  conditions_json jsonb NOT NULL,
  is_dynamic      boolean NOT NULL DEFAULT true
);

CREATE TABLE audiences (             -- רשימות חופשיות בעלות שם: "לידים וובינר 10.2.25"
  id             uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name           text NOT NULL,
  type           text NOT NULL,           -- 'segment' | 'static'
  segment_id     uuid REFERENCES segments(id),
  snapshot_count int                     -- ספירה ברגע השמירה (לקהל סטטי)
);

-- ── קמפיינים + מסמך המייל מהעורך הויזואלי ──
CREATE TABLE campaigns (
  id           uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name         text NOT NULL,
  subject      text NOT NULL,
  preheader    text,
  audience_id  uuid REFERENCES audiences(id),
  email_doc_id uuid REFERENCES email_docs(id),
  status       text NOT NULL DEFAULT 'draft',  -- draft | scheduled | sent
  scheduled_at timestamptz,
  sent_at      timestamptz,
  sender_name  text,
  sender_email text
);

CREATE TABLE email_docs (             -- מסמך הבלוקים שיוצר העורך הויזואלי המשותף
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  blocks_json jsonb NOT NULL,          -- אותו מודל בלוקים של עורך המרצים
  created_at  timestamptz NOT NULL DEFAULT now()
);

-- ── מנוע הסדרות (אוטומציה) ──
CREATE TABLE sequences (
  id                  uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name                text NOT NULL,
  trigger_event       text NOT NULL,
  trigger_config_json jsonb,
  active              boolean NOT NULL DEFAULT true
);

CREATE TABLE sequence_steps (
  id           uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  sequence_id  uuid REFERENCES sequences(id) ON DELETE CASCADE,
  position     int NOT NULL,
  type         text NOT NULL,           -- 'email' | 'wait' | 'condition'
  email_doc_id uuid REFERENCES email_docs(id),
  wait_amount  int,
  wait_unit    text,                  -- 'hours' | 'days'
  condition_json jsonb               -- בדיקת open/click של צעד קודם
);

CREATE TABLE sequence_enrollments (
  id           uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  sequence_id  uuid REFERENCES sequences(id),
  contact_id   uuid REFERENCES contacts(id),
  current_step int NOT NULL DEFAULT 0,
  status       text NOT NULL DEFAULT 'active',  -- active | done | exited
  next_run_at  timestamptz,
  UNIQUE (sequence_id, contact_id)        -- מונע הרשמה כפולה
);

-- ── יומן שליחות (אנליטיקה ומעקב) ──
CREATE TABLE email_sends (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  campaign_id      uuid REFERENCES campaigns(id),
  sequence_step_id uuid REFERENCES sequence_steps(id),
  contact_id       uuid REFERENCES contacts(id),
  status           text NOT NULL DEFAULT 'queued',
  -- queued | sent | opened | clicked | bounced | unsubscribed
  sent_at    timestamptz,
  opened_at  timestamptz,
  clicked_at timestamptz
);

מקור אמת יחיד: contacts ו‑tags משותפים עם ה‑CRM הקיים ("כל אנשי הקשר"). מערכת הדיוור אינה מנהלת רשימת אנשי קשר נפרדת — היא קוראת וכותבת לאותן טבלאות. תיוג שנעשה ב‑CRM משפיע מיד על הסגמנטים בדיוור, ולהיפך.

03מנוע התיוג (ידני + אוטומטי)

לכל איש קשר יכולות להיות תגיות ידניות ותגיות אוטומטיות. שתי הדרכים נכתבות ל‑contact_tags, ונבדלות ב‑added_by.

תיוג ידני

מנהל בוחר איש קשר ב‑CRM ("כל אנשי הקשר") ומוסיף/מסיר תגיות. הרשומה נכתבת עם added_by='manual'. זהו תיוג נקודתי, לא חוזר.

מנוע הכללים האוטומטי (event-driven)

אירועים נפלטים מרחבי המערכת — webinar_registered, cart_abandoned, lead_created, visited_n_times ועוד — ונצרכים ע"י worker שמעריך את tag_rules הפעילים ומכניס contact_tags(added_by='rule'). אין polling: ה‑worker מאזין לתור אירועים.

// worker: צריכת אירוע → הערכת כללים → תיוג
async function onEvent(evt: DomainEvent) {
  const rules = await getActiveRules(evt.name);  // trigger_event = evt.name
  for (const rule of rules) {
    if (!matches(rule.conditionsJson, evt)) continue;
    await addContactTag({
      contactId: evt.contactId,
      tagId:     rule.tagId,
      addedBy:   'rule',
      ruleId:    rule.id,
    });  // idempotent: PK (contact_id, tag_id)
  }
}

// דוגמה: כלל "וובינר"
// trigger_event: 'webinar_registered' → tag_id של 'וובינר'

הבחנה ויזואלית: תגיות אוטומטיות מסומנות בממשק באייקון ברק (kind='auto' / added_by='rule') כדי להבדיל אותן מתגיות ידניות. מנהל לא יכול לערוך תגית אוטומטית ידנית — רק את הכלל שמייצר אותה.

04סגמנטים וקהלים

סגמנט הוא conditions_json שמתקמפל ל‑SQL WHERE. הספירה החיה היא SELECT count(*) על אותו תנאי. קהל (audience) הוא רשימה שמורה בעלת שם, שיכולה להיות דינמית (סגמנט) או סטטית (snapshot).

// conditions_json — מערך תנאים + לוגיקה
{
  "logic": "and",
  "conditions": [
    { "field": "status",     "op": "eq",       "value": "lead" },
    { "field": "tag",        "op": "has",      "value": "וובינר" },
    { "field": "created_at", "op": "gte",      "value": "2026-02-01" }
  ]
}

// קומפילציה → WHERE
// status = 'lead' AND EXISTS (... tag 'וובינר') AND created_at >= '2026-02-01'
// ספירה חיה:  SELECT count(*) FROM contacts c WHERE <compiled>

קהל סטטי מול דינמי

  • קהל סטטי (type='static'): snapshot של מזהי אנשי הקשר ברגע השמירה. snapshot_count מקובע. שימושי ל"לידים וובינר 10.2.25" — רשימה קפואה שלא משתנה.
  • קהל דינמי (type='segment'): מצביע ל‑segment_id ומחושב מחדש בכל פעם — כולל ברגע השליחה.

שים לב: קהלים דינמיים מוערכים ברגע השליחה, ולכן הספירה יכולה להשתנות בין רגע השמירה לרגע השליחה (אנשי קשר נכנסו/יצאו מהסגמנט). הצג למשתמש את הספירה המעודכנת לפני אישור סופי.

05שליחת קמפיין בודד (Trigger.dev)

הזרימה: בחירת קהל → בניית המייל בעורך המשותף (email_docs) → תזמון → עבודת send-campaign ב‑Trigger.dev מפענחת את הקהל, מרנדרת כל מייל עם snapshot משתנים פר‑איש‑קשר, מכניסה רשומות ל‑email_sends ושולחת דרך ה‑ESP (למשל Resend / SES).

// trigger/admin/send-campaign.ts
export const sendCampaign = task({
  id: "send-campaign",
  run: async (payload: { campaignId: string }) => {
    const c   = await getCampaign(payload.campaignId);
    const doc = await getEmailDoc(c.emailDocId);
    const contacts = await resolveAudience(c.audienceId);  // דינמי → בזמן שליחה

    for (const batch of chunk(contacts, 200)) {       // batching + rate-limit
      await Promise.all(batch.map(async (ct) => {
        const vars = snapshotVars(ct);                 // {{first_name}} → ערך כעת
        const html = renderEmail(doc.blocksJson, vars);  // + בלוק הסרה חובה
        await recordSend({ campaignId: c.id, contactId: ct.id, status: "queued" });
        await esp.send({ to: ct.email, from: c.senderEmail, subject: c.subject, html });
      }));
      await wait.for({ seconds: 1 });                  // rate-limit מול ה-ESP
    }
    await markSent(c.id);
  },
});

חובה: בלוק "הסרה מרשימת התפוצה" מוזרק לכל מייל ב‑renderEmail גם אם המנהל הסיר אותו בעורך. ה‑rate-limit מול ה‑ESP נשמר דרך batching + wait.for.

06מנוע הסדרות (אוטומציה)

סדרה = טריגר + צעדים מסודרים (email / wait / condition). כשהטריגר נורה עבור איש קשר → נוצר sequence_enrollment. עבודת Trigger.dev מתוזמנת מקדמת את ההרשמות צעד‑צעד.

// trigger/admin/advance-enrollments.ts — רץ כל מספר דקות
export const advanceEnrollments = schedules.task({
  id: "advance-enrollments",
  cron: "*/5 * * * *",
  run: async () => {
    const due = await getDueEnrollments();  // status='active' AND next_run_at <= now()
    for (const e of due) {
      const step = await getStep(e.sequenceId, e.currentStep);
      if (!step) { await finish(e); continue; }

      switch (step.type) {
        case "email": {
          const doc  = await getEmailDoc(step.emailDocId);
          const vars = snapshotVars(e.contact);
          await esp.send({ to: e.contact.email, html: renderEmail(doc.blocksJson, vars) });
          await advance(e);  // current_step++ ; next_run_at = now()
          break;
        }
        case "wait":
          await advance(e, { nextRunAt: addTime(now(), step.waitAmount, step.waitUnit) });
          break;
        case "condition": {                          // בדיקת open/click של מייל קודם
          const ok = await evalCondition(step.conditionJson, e);
          await branch(e, ok);  // קפיצה לצעד מתאים
          break;
        }
      }
    }
  },
});

אידמפוטנטיות: כל קידום מתבצע בטרנזקציה על current_step כדי שריצה כפולה לא תשלח פעמיים. ה‑UNIQUE(sequence_id, contact_id) מונע הרשמה כפולה של אותו איש קשר לאותה סדרה.

07שימוש חוזר בעורך + רינדור

צעד התוכן של הקמפיין וכל מייל בסדרה ("ערוך מייל") פותחים את עורך מייל ויזואלי.html — אותו עורך בלוקים בדיוק שבו משתמשים המרצים. אין עורך שני, אין registry שני.

  • registry משותף: אותם סוגי בלוקים (טקסט, תמונה, כפתור CTA, מפריד, פוטר, הסרת תפוצה) — מוגדרים פעם אחת ומשמשים את שני המודולים.
  • מודל אחיד: פלט העורך הוא blocks_json שנשמר ב‑email_docs — אותו מבנה למרצה ולהנהלה.
  • רינדור: בעת שליחה, blocks_json מתורגם ל‑MJML/HTML עם styles inline; משתנים נפתרים פר‑נמען (snapshot); בלוקי הסרה וכתובת פיזית נאכפים.
// קישור "ערוך מייל" — אותו עורך של המרצים
const editorUrl = `עורך מייל ויזואלי.html?docId=${emailDocId}`;

// רינדור משותף: blocks_json → HTML email
function renderEmail(blocks: Block[], vars: Vars): string {
  const withFooter = ensureUnsubscribeBlock(blocks);   // חובה חוקית
  const mjml = blocksToMjml(withFooter, registry);
  return resolveVars(mjml2html(mjml).html, vars); // {{first_name}} → ערך הנמען
}

Deliverability: דומיין השולח חייב SPF / DKIM / DMARC מוגדרים נכון, וכל מייל יוצא עם List-Unsubscribe header (וגם הבלוק בגוף). זה מבטיח שהקמפיינים לא ינחתו בספאם ושומר על מוניטין הדומיין.
